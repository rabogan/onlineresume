Active Robot

  • Read integers n and m, then read in a matrix (if necessary)
  • Read in an integer K, then add K commands. Each command is two values
    • Direction from 1 to 4: up, right, down, left
    • Steps: a number to number steps to take in the direction. Steps [1, 1000000000]
    • If the robot hits the wall during the move, it circulates in the matrix
    • For every command, print where the robot is now
  • Input
    • 3 4   4   2 1   3 2   4 2   1 3
    • In the command 2 1, 2 indicates the direction (right), and 1 is the number of steps to take in that direction.
  • Output
    •          (0,1) (2,1) (2,3) (2,3)
Back 

  #include <iostream>
  using namespace std;
  //active robot!
  int main()
  {
  	int my_arr[100][100];

  	int n = 0;
  	int m = 0;
  	cin >> n >> m;
  	cout << "Enter your array!\n";
  	for (int i = 0; i < n; ++i)
  	{
  		for (int j = 0; j < m; ++j)
  		{
  			cin >> my_arr[i][j];
  		}
  	}

  	cout << "Enter your k, then the k-commands\n";
  	int k = 0;
  	cin >> k;

  	int dir_row[4] = { -1,0,1,0 };
  	int dir_col[4] = { 0,1,0,-1 };

  	int start_row = 0;
  	int start_col = 0;

  	while (k--)
  	{
  		int k_direction = 0;
  		int k_steps = 0;
  		cout << "Enter k direction: (1-up, 2-right, 3-down, 4-left), followed by the number of steps:\n";
  		cin >> k_direction >> k_steps;

  		k_direction--;

  		for (int i = 0; i < k_steps; ++i)
  		{
  			start_row += dir_row[k_direction];
  			start_col += dir_col[k_direction];

  			if (start_row >= n)
  				start_row = 0;

  			if (start_row < 0)
  				start_row = n - 1;

  			if (start_col >= m)
  				start_col = 0;

  			if (start_col < 0)
  				start_col = m - 1;
  		}
  		cout << "(" << start_row << ',' << start_col << ")" << endl;
  	}
  }
