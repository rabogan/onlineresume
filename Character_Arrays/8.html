How many primes?

  • Read in the integers n and m, then read a 2D matrix n*m. Then read in an integer Q, for Q queries. Each queries is a grid from the top left (i,j) and #rows and #columns.
    • Read in four integers for i j rows columns.
  • For each query, print how many prime numbers are in the requested grid
  • Input → Output
    • 3 4
    • 8 2 9 5
    • 3 2 27 6
    • 7 8 29 22
    • 2
    • 1 0 2 2 → 3 (primes 3, 2, 7 in rectangle (0,1) (2 1))
    • 0 1 2 3 → 3 (primes 2, 5, 2 in rectangle (0,1) (1,3))
Back 
  
  #include <iostream>
  using namespace std;

  bool is_prime(int x, int tester = 3)
  {
    if (x <= 1)
      return false;

    if (x == 2)
      return true;

    if (x == tester)
      return true;

    if (x % tester == 0)
      return false;

    is_prime(x, tester + 1);
  }

  int main()
  {
    int my_arr[100][100];

    int n = 0;
    int m = 0;

    cin >> n >> m;
    cout << "Enter your array values!\n";
    for (int i = 0; i < n; ++i)
    {
      for (int j = 0; j < m; ++j)
      {
        cin >> my_arr[i][j];
      }
    }
    cout << "Enter your q\n";
    int q = 0;
    cin >> q;

    while (q--)
    {
      cout << "What coordinates do you want to start from?\n";
      int start_row = 0;
      int start_col = 0;

      cin >> start_row >> start_col;

      cout << "How many rows and columns?\n";
      int total_rows = 0;
      int total_cols = 0;
      cin >> total_rows >> total_cols;

      int prime_count = 0;
      for (int i = start_row; i < start_row + total_rows; ++i)
      {
        for (int j = start_col; j < start_col + total_cols; ++j)
        {
          if (i >= n || j >= m)
            continue;
          int digit = my_arr[i][j];
          cout << digit << ' ';

          if (is_prime(digit))
          {
            prime_count++;
          }

        }
        cout << endl;
      }
      cout << "The prime count is " << prime_count << endl;
    }
    return 0;
  }
